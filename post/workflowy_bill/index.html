// ==UserScript==
// @name         Workflowy Figma Embed
// @namespace    http://tampermonkey.net/
// @version      0.1
// @description  Embed Figma files in Workflowy
// @author       You
// @match        https://workflowy.com/*
// @grant        none
// ==/UserScript==

(function() {
    'use strict';

    const figmaFormat = {
        regex: /https?:\/\/(www\.)?figma\.com\/(file|proto)\/([0-9a-zA-Z]+)(\/.*)?/i,
        embed: 'https://www.figma.com/embed?embed_host=share&url=https%3A%2F%2Fwww.figma.com%2F$2%2F$3'
    };

    const checkForEmbeds = () => {
        document.querySelectorAll('a:not([data-processed])').forEach(a => {
            const url = a.href;
            const match = url.match(figmaFormat.regex);
            if (match) {
                const embedUrl = url.replace(figmaFormat.regex, figmaFormat.embed);
                const iframe = document.createElement('iframe');
                iframe.src = embedUrl;
                iframe.width = 800;
                iframe.height = 600;
                iframe.style.border = '1px solid rgba(0, 0, 0, 0.1)';
                a.parentElement.insertBefore(iframe, a.nextSibling);
                a.setAttribute('data-processed', true);
            }
        });
    };

    setInterval(checkForEmbeds, 500);
})();
